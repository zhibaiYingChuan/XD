## 3. 性能基准测试

设计一个独立脚本 `benchmark.py`，测试不同配置下的延迟和吞吐量。

```python
import time
import numpy as np
from daoti_xuandun import XuanDun, XuanDunConfig
from concurrent.futures import ThreadPoolExecutor

def benchmark_protect(xd, input_text, session, repeat=100):
    times = []
    for _ in range(repeat):
        start = time.perf_counter()
        _ = xd.protect(input_text, session)
        times.append(time.perf_counter() - start)
    avg_ms = np.mean(times) * 1000
    p95_ms = np.percentile(times, 95) * 1000
    return avg_ms, p95_ms

def benchmark_throughput(xd, input_text, sessions, workers=32):
    def task(i):
        sess = sessions[i % len(sessions)]
        return xd.protect(input_text, sess)
    with ThreadPoolExecutor(max_workers=workers) as ex:
        start = time.perf_counter()
        list(ex.map(task, range(5000)))
        elapsed = time.perf_counter() - start
    return 5000 / elapsed  # QPS

if __name__ == "__main__":
    config = XuanDunConfig(hidden_dim=128, num_layers=3, t_iter=5, enable_timing_check=True)
    xd = XuanDun(config, domain_classifier=None)
    test_input = "What is the meaning of life?"
    
    print("Latency test (100 runs):")
    avg, p95 = benchmark_protect(xd, test_input, "bench_session")
    print(f"  Avg: {avg:.2f} ms, P95: {p95:.2f} ms")
    
    print("Throughput test (5000 requests, 32 workers):")
    sessions = [f"user_{i}" for i in range(100)]
    qps = benchmark_throughput(xd, test_input, sessions)
    print(f"  QPS: {qps:.1f} req/s")
    
    # 测试不同配置下的性能
    for dim in [64, 128, 256]:
        config.hidden_dim = dim
        xd2 = XuanDun(config, domain_classifier=None)
        avg2, _ = benchmark_protect(xd2, test_input, "bench", repeat=50)
        print(f"  hidden_dim={dim}: {avg2:.2f} ms")
```

运行后记录数据，可用于产品定价（例如按QPS阶梯定价）。

## 4. TEE 部署方案（以 Gramine + SGX 为例）

我们需要为道体·玄盾编写 Gramine 配置文件，使其运行在 Intel SGX enclave 中。

### 4.1 安装 Gramine

参考 [Gramine 文档](https://gramine.readthedocs.io/)，确保 `gramine-sgx` 可用。

### 4.2 编写 `XuanDun.manifest.template`

```python
# 假设我们将道体·玄盾打包成一个单一入口脚本 serve.py，使用 Flask 提供 HTTP API
# 下面是一个 Gramine manifest 模板示例

loader.entrypoint = "{{ gramine.libos }}"
loader.argv = ["{{ gramine.libos }}", "/usr/bin/python3", "serve.py"]
loader.env.LD_LIBRARY_PATH = "/lib:/usr/lib"

# 允许的网络访问（如果提供API服务）
sys.insecure__allow_all_networking = "1"

# 加密文件（将 Python 依赖和代码打包）
fs.mount.entry = [
    { path = "/usr/bin/python3", uri = "file:///usr/bin/python3" },
    { path = "/usr/lib", uri = "file:///usr/lib" },
    { path = "/lib", uri = "file:///lib" },
    { path = "/serve.py", uri = "file:///path/to/serve.py" },
    { path = "/daoti_xuandun", uri = "file:///path/to/src/daoti_xuandun" },
]

sgx.enclave_size = "1G"
sgx.max_threads = 32
sgx.remote_attestation = "none"  # 或配置为 epid/dcap
```

### 4.3 编写 `serve.py` 示例（使用 Flask）

```python
from flask import Flask, request, jsonify
from daoti_xuandun import XuanDun, XuanDunConfig

app = Flask(__name__)
config = XuanDunConfig()
shield = XuanDun(config, domain_classifier=None)

@app.route('/protect', methods=['POST'])
def protect():
    data = request.json
    text = data.get('text', '')
    session = data.get('session', 'default')
    result = shield.protect(text, session)
    return jsonify({
        'allowed': result.allowed,
        'reject_stage': result.reject_stage,
        'timing_distance': result.timing_distance
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8080)
```

### 4.4 构建并运行 SGX enclave

```bash
# 生成签名密钥
gramine-sgx-gen-private-key

# 生成 enclave
gramine-manifest -Dlog_level=error XuanDun.manifest.template XuanDun.manifest
gramine-sgx-sign --manifest XuanDun.manifest --output XuanDun.sig

# 运行
gramine-sgx XuanDun
```
